60/*
61 * Initialisierung von WiringPi und der Segmentanzeige.
62 * Muss genau einmal vor der Benutzung der Anzeige aufgerufen werden.
63 */
64int TM1637_setup() {
65 static int device_initialised = 0;
66
67 if (!device_initialised) {
68 /* Versuche WiringPi zu initialisieren */
69 if (wiringPiSetup() == -1) {
70 fprintf(stderr, "setup wiringPi failed ! n");
71 return EXIT_FAILURE;
72 }
73
74 /* Jetzt noch die Segmentanzeige initialisieren */
75 /* beide Pins als Output konfigurieren */
76 pinMode(PIN_CLOCK, OUTPUT);
77 pinMode(PIN_DATA, OUTPUT);
78
79 /* Vorbereitung für start(): beide Pins auf 1 */
80 digitalWrite(PIN_CLOCK, HIGH);
81 digitalWrite(PIN_DATA, HIGH);
82 delayMicroseconds(DELAY_TIMER);
83
84 /* Anzeige löschen */
85 TM1637_clear_display();
86
87 /* Initialisierung nicht mehrfach durchfuehren*/
88 device_initialised = 1;
89
90 return EXIT_SUCCESS;
91 }
92
93 /* FAILURE falls bereits initialisiert */
94 return EXIT_FAILURE;
95}

97/*
98 * Low-Level Funktion zum Abschliessen des Datentransfers.
99 * Muss am Ende von TM1637_write_byte() aufgerufen werden.
100 */
101void TM1637_ack() {
102 int count = 0;
103
104 /*
105 * Starte ack():
106 * CLK auf 0, DIO auf 1 (wird durch den Chip heruntergezogen)
107 */
108 digitalWrite(PIN_CLOCK, LOW);
109 delayMicroseconds(DELAY_TIMER);
110
111 /* DIO als Input konfigurieren (wir wollen die 0 vom runterziehen lesen) */
112 digitalWrite(PIN_DATA, HIGH);
113 pinMode(PIN_DATA, INPUT);
114 delayMicroseconds(DELAY_TIMER);
115
116 /* CLK auf 1: Warte auf das ACK vom Chip */
117 digitalWrite(PIN_CLOCK, HIGH);
118 delayMicroseconds(DELAY_TIMER);
119
120 /* versuche die 0 auf dem DIO zu lesen */
121 while (digitalRead(PIN_DATA)) {
122 count += 1;
123 if (count == 200) {
124 break; /* wir geben auf */
125 }
126 }
127
128 /* CLK und DIO auf 0: Ausgangspunkt fuer stop */
129 digitalWrite(PIN_CLOCK, LOW);
130 delayMicroseconds(DELAY_TIMER);
131
132 /* DIO-Pin restaurieren */
133 pinMode(PIN_DATA, OUTPUT);
134 digitalWrite(PIN_DATA, LOW);
135 delayMicroseconds(DELAY_TIMER);
136}

159/*
160 * Low-Level Funktionen zum Setzen eines Segments.
161 * Der Parameter data ist eine Bitsequenz nach NUMBER_TO_BYTE, ohne den Dezimalpunkt.
162 */
163void TM1637_display(segment seg, byte data, dot point) {
164 byte seg_data;
165
166 seg_data = data;
167 if (point == ON) {
168 seg_data = seg_data + DOT_OFFSET;
169 }
170
171 /* CMD1: Modus setzen */
172 TM1637_start();
173 TM1637_write_byte(ADDR_FIXED);
174 TM1637_stop();
175
176 /* CMD2: Daten fuer Segment schicken */
177 TM1637_start();
178 TM1637_write_byte(seg | STARTADDR);
179 TM1637_write_byte(seg_data);
180 TM1637_stop();
181
182 /* CMD3: Helligkeit setzen */
183 TM1637_start();
184 TM1637_write_byte(BRIGHT_OFFSET + bright);
185 TM1637_stop();
186}

189 * High-Level Funktionen zum Setzen eines Segments.
190 * Der Parameter number ist eine Gleitkommazahl zwischen -99.9 und 999.9 und
191 * wird auf der Segmentanzeige mit drei Vorkomma- und einer Nachkommstelle
192 * angezeigt.

194void TM1637_display_number(float number) {
195 TM1637_display(SEG1, TM1637_calculate_display(number, SEG1), OFF);
196 TM1637_display(SEG2, TM1637_calculate_display(number, SEG2), OFF);
197 TM1637_display(SEG3, TM1637_calculate_display(number, SEG3), ON);
198 TM1637_display(SEG4, TM1637_calculate_display(number, SEG4), OFF);

201/*
202 * Hilfsfunktion: Berechnet zu number die Bitsequenz für die Stelle seq.
203 * Wird in TM1637_display_number() genutzt.
204 */
205byte TM1637_calculate_display(float number, segment seg) {
206 static int lastnumber = 0;
207 static byte segmente[] = { 0x3f, 0x3f, 0x3f, 0x3f };
208 unsigned char digit;
209 int reminder;
210
211 /* Range Check */
212 if (number > 999.9f) {
213 fprintf(stderr,
214 "calculateDisplay does not accept values higher than 999.9 \n");
215 return 0x00; /* Error value */
216 }
217 if (number < -99.9f) {
218 fprintf(stderr,
219 "calculateDisplay does not accept values lower than -99.9 \n");
220 return 0x00; /* Error value */
221 }
222
223 reminder = (int) (number * 10.0);
224 if (reminder != lastnumber) {
225 /* Neue Zahl ... alles neu berechnen */
226 lastnumber = reminder;
227
228 /* Vorzeichenbehandlung */
229 if (reminder >= 0) {
230 /* calculate first segment and subtract value from the rest of the value to be displayed */
231 digit = reminder / 1000;
232 reminder = reminder - digit * 1000;
233 segmente[0] = NUMBER_TO_BYTE[digit];
234 } else {
235 reminder = abs(reminder);
236
237 /* first segment displays a "-" */
238 segmente[0] = 0x40;
239 }
240
241 /* calculate second segment and subtract value from the rest of the value to be displayed */
242 digit = reminder / 100;
243 reminder = reminder - digit * 100;
244 segmente[1] = NUMBER_TO_BYTE[digit];
245
246 /* calculate third segment and subtract value from the rest of the value to be displayed */
247 digit = reminder / 10;
248 reminder = reminder - digit * 10;
249 segmente[2] = NUMBER_TO_BYTE[digit];
250
251 /* calculate fourth segment and subtract value from the rest of the value to be displayed */
252 segmente[3] = NUMBER_TO_BYTE[reminder];
253 }
254
255 /* Rueckgabe */
256 switch (seg) {
257 case SEG1:
258 return segmente[0];
259 case SEG2:
260 return segmente[1];
261 case SEG3:
262 return segmente[2];
263 case SEG4:
264 return segmente[3];
265 default:
266 return 0x00; /* Error value */
267 }
268}

273static void TM1637_start(void) {
274 /* beide Pins muessen bei 1 starten
275 *
276 * Sollte nach init() bzw. nach stop() bereit der Fall sein,
277 * aber sicherheitshalber noch einmal einstellen
278 */
279 digitalWrite(PIN_CLOCK, HIGH);
280 digitalWrite(PIN_DATA, HIGH);
281 delayMicroseconds(DELAY_TIMER);
282
283 /* DIO muss auf 0 gehen => Start */
284 digitalWrite(PIN_DATA, LOW);
285 delayMicroseconds(DELAY_TIMER);
286
287 /* CLK muss ebenfalls auf 0 gehen */
288 digitalWrite(PIN_CLOCK, LOW);
289 delayMicroseconds(DELAY_TIMER);
290}
